            if (!sceneConfig) return;
            
            // 获取图片配置（如果是数组则轮换）
            let imageConfig;
            if (Array.isArray(sceneConfig)) {
                let index;
                if (sceneNum === 3) {
                    index = scene3Index % sceneConfig.length;
                    scene3Index++;
                } else if (sceneNum === 4) {
                    index = scene4Index % sceneConfig.length;
                    scene4Index++;
                } else if (sceneNum === 5) {
                    index = scene5Index % sceneConfig.length;
                    scene5Index++;
                } else {
                    index = 0;
                }
                imageConfig = sceneConfig[index];
            } else {
                imageConfig = sceneConfig;
            }
            
            // 设置图片和描述
            stageImage.src = imageConfig.image;
            imageDesc.textContent = imageConfig.desc;
            imageDesc.style.borderColor = imageConfig.color;
            imageContainer.style.borderColor = imageConfig.color;
            
            // 清除之前的动感效果
            stageImage.className = "stage-image";
            
            // 应用新的动感效果
            setTimeout(() => {
                if (imageConfig.effect) {
                    stageImage.classList.add(imageConfig.effect);
                }
            }, 100);
            
            // 观看人数增加
            viewerCount += Math.floor(Math.random() * 10) + 5;
            viewerCountEl.textContent = viewerCount;
        }
        
        function handleScene6() {
            // 第六幕：镜镜被操烂
            shameLevel = 100; // 羞耻度满值
            updateShameLevel();
            
            addDanmaku("主播被操烂了！", 6);
            setTimeout(() => addDanmaku("镜镜被操得稀烂", 6), 1000);
        }
        
        function handleScene7() {
            // 第七幕：尤金宣告
            addDanmaku("尤金怒视镜头！", 7);
            setTimeout(() => addDanmaku("占有欲好强，舍不得结束", 7), 1000);
            setTimeout(() => addDanmaku("还想继续看镜镜被操", 7), 2000);
            
            // 5秒后屏幕黑掉
            setTimeout(() => {
                blackoutScreen();
            }, 5000);
        }
        
        function blackoutScreen() {
            // 屏幕逐渐黑掉
            const overlay = document.createElement('div');
            overlay.style.position = 'fixed';
            overlay.style.top = '0';
            overlay.style.left = '0';
            overlay.style.width = '100%';
            overlay.style.height = '100%';
            overlay.style.background = '#000';
            overlay.style.zIndex = '100';
            overlay.style.opacity = '0';
            overlay.style.transition = 'opacity 2s ease';
            document.body.appendChild(overlay);
            
            setTimeout(() => {
                overlay.style.opacity = '1';
            }, 100);
            
            // 最后显示重置按钮
            setTimeout(() => {
                scene7Btn.style.display = "none";
                resetBtn.style.display = "inline-block";
            }, 2500);
        }
        
        // ==================== 打赏系统 ====================
        function enableGiftButtons() {
            roseBtn.disabled = false;
            carBtn.disabled = false;
            planeBtn.disabled = false;
            rocketBtn.disabled = false;
        }
        
        function sendGift(type) {
            const gift = GIFTS[type];
            if (!gift) return;
            
            giftCount++;
            giftCountEl.textContent = giftCount;
            
            // 增加羞耻度
            shameLevel += gift.shame;
            if (shameLevel > 100) shameLevel = 100;
            updateShameLevel();
            
            // 增加高潮次数
            orgasmCount += gift.orgasm;
            orgasmCountEl.textContent = orgasmCount;
            
            // 打赏特效
            const effect = document.createElement('div');
            effect.className = 'gift-effect';
            effect.textContent = `${gift.emoji} ${gift.name}`;
            effect.style.color = gift.color;
            document.body.appendChild(effect);
            
            effect.animate([
                { opacity: 0, transform: 'translate(-50%, -50%) scale(0.5)' },
                { opacity: 1, transform: 'translate(-50%, -50%) scale(1.2)' },
                { opacity: 0, transform: 'translate(-50%, -50%) scale(0.7)' }
            ], {
                duration: 2000,
                easing: 'ease-out'
            }).onfinish = () => effect.remove();
            
            // 打赏弹幕
            const users = ["老板A", "大哥B", "土豪C", "匿名老板", "镜镜爱好者"];
            const user = users[Math.floor(Math.random() * users.length)];
            addDanmaku(`🎁 ${user} 送出了${gift.name} - ${gift.action}`);
            
            // 主播反应
            const reactions = [
                "啊！羞耻...",
                "不要...太羞耻了...",
                "继续...好羞耻...",
                "被这样打赏好羞耻..."
            ];
            if (Math.random() > 0.5) {
                setTimeout(() => addDanmaku(`💬 镜镜: "${reactions[Math.floor(Math.random() * reactions.length)]}"`), 500);
            }
            
            // 观看人数增加
            viewerCount += Math.floor(Math.random() * 8) + 3;
            viewerCountEl.textContent = viewerCount;
        }
        
        // ==================== 羞耻度系统 ====================
        function updateShameLevel() {
            shameValueEl.textContent = `${shameLevel}%`;
            shameFill.style.width = `${shameLevel}%`;
            
            // 羞耻度超过50时弹幕特殊反应
            if (shameLevel >= 50 && shameLevel < 80) {
                if (Math.random() > 0.7) {
                    addDanmaku("镜镜羞耻度好高！", currentScene);
                }
            } else if (shameLevel >= 80) {
                if (Math.random() > 0.7) {
                    addDanmaku("羞耻度爆表！镜镜崩溃了！", currentScene);
                }
            }
        }
        
        // ==================== 工具函数 ====================
        function getSceneTitle(sceneNum) {
            const titles = {
                2: "第二幕！等待调教！",
                3: "第三幕！操穴开始！",
                4: "第四幕！深度调教！", 
                5: "第五幕！高潮崩坏！",
                6: "主播被操烂了！",
                7: "尤金宣告！"
            };
            return titles[sceneNum] || `第${sceneNum}幕开始`;
        }
        
        function updateControls() {
            // 隐藏所有场景按钮
            const sceneBtns = [scene2Btn, scene3Btn, scene4Btn, scene5Btn, scene6Btn, scene7Btn, resetBtn];
            sceneBtns.forEach(btn => btn.style.display = "none");
            
            // 显示当前场景对应的按钮
            if (currentScene === 1 && isLiveActive) {
                scene2Btn.style.display = "inline-block";
            } else if (currentScene === 2) {
                scene3Btn.style.display = "inline-block";
            } else if (currentScene === 3) {
                scene4Btn.style.display = "inline-block";
            } else if (currentScene === 4) {
                scene5Btn.style.display = "inline-block";
            } else if (currentScene === 5) {
                scene6Btn.style.display = "inline-block";
            } else if (currentScene === 6) {
                scene7Btn.style.display = "inline-block";
            }
        }
        
        function updateStats() {
            viewerCountEl.textContent = viewerCount;
            orgasmCountEl.textContent = orgasmCount;
            giftCountEl.textContent = giftCount;
        }
        
        function resetLive() {
            if (!confirm('确定要重置直播吗？')) return;
            
            // 重置状态
            currentScene = 1;
            viewerCount = 1;
            orgasmCount = 0;
            giftCount = 0;
            shameLevel = 0;
            danmakuCount = 0;
            scene3Index = 0;
            scene4Index = 0;
            scene5Index = 0;
            
            // 重置界面
            stageImage.src = "";
            stageImage.className = "stage-image";
            imageDesc.textContent = "等待直播开始...";
            imageDesc.style.borderColor = "#ff3366";
            imageContainer.style.borderColor = "#ff3366";
            
            // 清除黑屏覆盖
            const overlay = document.querySelector('div[style*="position: fixed"]');
            if (overlay) overlay.remove();
            
            // 重置控制栏
            startBtn.disabled = false;
            startBtn.style.display = "inline-block";
            scene2Btn.style.display = "none";
            scene3Btn.style.display = "none";
            scene4Btn.style.display = "none";
            scene5Btn.style.display = "none";
            scene6Btn.style.display = "none";
            scene7Btn.style.display = "none";
            resetBtn.style.display = "none";
            giftButtons.style.display = "none";
            
            // 禁用打赏按钮
            roseBtn.disabled = true;
            carBtn.disabled = true;
            planeBtn.disabled = true;
            rocketBtn.disabled = true;
            
            // 更新数据
            updateStats();
            updateShameLevel();
            
            // 清除弹幕
            danmakuLayer.innerHTML = '';
            if (danmakuInterval) clearInterval(danmakuInterval);
            
            // 初始弹幕
            setTimeout(() => {
                addDanmaku("直播重置，准备重新开始", 1);
            }, 500);
            
            isLiveActive = false;
        }
        
        // ==================== 弹幕系统 ====================
        function startDanmaku() {
            // 定时生成情境化弹幕
            danmakuInterval = setInterval(() => {
                if (isLiveActive) {
                    for (let i = 0; i < Math.floor(Math.random() * 2) + 1; i++) {
                        addDanmaku(null, currentScene);
                    }
                }
            }, 2500);
        }
        
        function addDanmaku(text, sceneNum = currentScene) {
            danmakuCount++;
            
            const danmaku = document.createElement('div');
            danmaku.className = 'danmaku';
            
            // 如果没有提供文本，从对应场景的弹幕库中随机选择
            if (!text) {
                const sceneKey = `scene${sceneNum}`;
                const danmakuPool = DANMAKU_LIB[sceneKey] || DANMAKU_LIB.scene1;
                text = danmakuPool[Math.floor(Math.random() * danmakuPool.length)];
            }
            
            danmaku.textContent = text;
            
            // 随机位置
            const top = Math.random() * 85;
            danmaku.style.top = `${top}%`;
            danmaku.style.left = '100%';
            
            // 随机颜色
            const colors = ['#ff6666', '#66aaff', '#66ff99', '#ffff66', '#ff66cc', '#8b5cf6'];
            danmaku.style.color = colors[Math.floor(Math.random() * colors.length)];
            
            danmakuLayer.appendChild(danmaku);
            
            // 动画
            const width = window.innerWidth + danmaku.offsetWidth + 100;
            const duration = 7000 + Math.random() * 6000;
            
            danmaku.animate([
                { transform: 'translateX(0)' },
                { transform: `translateX(-${width}px)` }
            ], {
                duration: duration,
                easing: 'linear'
            }).onfinish = () => danmaku.remove();
        }
